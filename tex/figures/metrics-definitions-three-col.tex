\newcommand{\SweepsMetricName}{
Number of coalescence events
}
\newcommand{\SweepsMetricMeasurement}{
The number of coalescence events that have occurred along a given lineage
}
\newcommand{\SweepsMetricMeaning}{
Frequency of selective sweeps in a population
}


\newcommand{\MutationCountMetricName}{
Mutation count
}
\newcommand{\MutationCountMetricMeasurement}{
The number of mutations that have occurred along a lineage
}
\newcommand{\MutationCountMetricMeaning}{

}

\newcommand{\PhenotypicVolatilityMetricName}{
Phenotypic volatility
}
\newcommand{\PhenotypicVolatilityMetricMeasurement}{
Number of offspring along the lineage with a different phenotypic profile than their parent
}
\newcommand{\PhenotypicVolatilityMetricMeaning}{
Rate at which genetic changes affect phenotype
}

\newcommand{\MutationalStabilityMetricName}{
Mutational stability
}
\newcommand{\MutationalStabilityMetricMeasurement}{
Proportion of \textit{mutated} offspring along the lineage with a different phenotypic profile than their parent
}
\newcommand{\MutationalStabilityMetricMeaning}{
Frequency at which mutations affect phenotype.
}

\newcommand{\TaskPerformanceMetricName}{
Final [novel] task count
}
\newcommand{\TaskPerformanceMetricMeasurement}{
Number of unique \textit{novel} tasks performed by most abundant genotype in final population (ranges from 0 to 71)
}
\newcommand{\TaskPerformanceMetricMeaning}{
Exploitation of fitness landscape (i.e., the mapping between genetic space and phenotype space).
}

\newcommand{\TaskDiscoveryMetricName}{
Lineage [novel] task count
}
\newcommand{\TaskDiscoveryMetricMeasurement}{
Number of unique \textit{novel} tasks performed by along a given lineage, even if tasks are eventually lost (ranges from 0 to 71)
}
\newcommand{\TaskDiscoveryMetricMeaning}{
Exploration of fitness landscape (i.e., the mapping between genetic space and phenotype space).
}

\newcommand{\TaskLossMetricName}{
Lineage [novel] task loss
}
\newcommand{\TaskLossMetricMeasurement}{
Number of times along a lineage that a \textit{novel} task was performed by a parent but not its offspring
}
\newcommand{\TaskLossMetricMeaning}{
How often a lineage fails to retain novel tasks (i.e., how well traits are maintained)
}

\newcommand{\FinalPoisonMetricName}{
Final \code{poison} count
}
\newcommand{\FinalPoisonMetricMeasurement}{
Number of times the \code{poison} instruction is executed by the most abundant genotype in the final population
}
\newcommand{\FinalPoisonMetricMeaning}{
Efficiency of a population in purging deleterious mutations
}

\newcommand{\LineagePoisonMetricName}{
Lineage \code{poison} increases
}
\newcommand{\LineagePoisonMetricMeasurement}{
Number of offspring along a given lineage that execute the \code{poison} instruction more often than their parent
}
\newcommand{\LineagePoisonMetricMeaning}{
How often deleterious hitchhiking occurred along the lineage
}
